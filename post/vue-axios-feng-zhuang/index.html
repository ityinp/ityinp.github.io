params跟data

在使用 axios 时，注意到配置选项中包含 params 和 data 两者，以为他们是相同的，实则不然。因为 params 是添加到 url 的请求字符串中的，用于 get 请求。而 data 是添加到请求体（body）中的， 用于 post 请求。

import http from 'axios'
import qs from 'qs'
import { Message, Loading } from 'element-ui'


export function get(url, data) {
  var corsUrl = "http://yinpnas.top:1234/"
  if (data == null || data == '') {
    data = {
      s: new Date().getTime()
    }
  } else {
    data.s = new Date().getTime()
  }

  const loading = Loading.service({
    lock: true, // lock的修改符--默认是false
    text: '正在拼命加载中，请稍后', // 显示在加载图标下方的加载文案
    background: 'rgba(0, 0, 0, 0.5)', // 遮罩层颜色
    target: document.querySelector('#table') // loadin覆盖的dom元素节点
  })
  return new Promise((res, err) => {
    http({
      method: 'get',
      url: corsUrl + url,
      parmas: data,
    })
      .then((response) => {
        res(response)
        // 成功回调函数停止加载
        loading.close()
      })
      .catch((error) => {
        Message.error('网络异常')
        err(error)
        // 成功回调函数停止加载
        loading.close()
      })
  })
}

export function post(url, data) {
  var corsUrl = "http://yinpnas.top:1234/"
  if (data == null || data == '') {
    data = {
      s: new Date().getTime()
    }
  } else {
    data.s = new Date().getTime()
  }

  const loading = Loading.service({
    lock: true, // lock的修改符--默认是false
    text: '正在拼命加载中，请稍后', // 显示在加载图标下方的加载文案
    background: 'rgba(0, 0, 0, 0.5)', // 遮罩层颜色
    target: document.querySelector('#table') // loadin覆盖的dom元素节点
  })
  return new Promise((res, err) => {
    http({
      method: 'post',
      url: corsUrl + url,
      data: qs.stringify(data),
    })
      .then((response) => {
        res(response)
        // 成功回调函数停止加载
        loading.close()
      })
      .catch((error) => {
        Message.error('网络异常')
        err(error)
        // 成功回调函数停止加载
        loading.close()
      })
  })
}
